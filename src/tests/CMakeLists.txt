# List of tests
set(TESTS clo_rng_test_old clo_scan_test)

# Common libs for tests

# Add a target for each utility

endforeach()

# How to install tests
install(TARGETS ${TESTS} RUNTIME DESTINATION bin)

add_custom_target(tests DEPENDS ${TESTS})
